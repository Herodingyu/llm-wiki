## 第10章 多核通信与同步机制

## 10.3 自旋锁、信号量多核实现

### 10.3.1 设计来源与设计思想

设计来源：自旋锁与信号量是现代多核系统中实现同步与互斥的基础机制，其设计思想源于并发控制理论在嵌入式多核环境中的具体实践：

1. 并发控制理论：Dijkstra的信号量概念、Lamport的分布式同步算法
2. 多核处理器架构：ARM Cortex-R52+的原子指令支持和内存一致性模型
3. 实时系统同步需求：汽车电子和工业控制要求确定性的同步延迟
4. 锁优化理论：排队锁、读写锁、RCU等高级锁机制
5. 功能安全标准：ISO 26262要求同步机制具备可预测性和故障安全特性

设计思想演进：从简单的忙等待到复杂的公平锁机制，体现了同步机制成熟化的过程：

![图片](https://mmbiz.qpic.cn/mmbiz_png/QTnjRdicb9qYWn0pKmOIwjUTIL95cuxvEUb4Gw2CtfSTWiaMLfrJRCdYVzl4FPVcuIVcbeu1Y8EiaFKviacFKdmqDg/640?wx_fmt=png&from=appmsg&tp=webp&wxfrom=5&wx_lazy=1#imgIndex=0)  

核心设计原则：

1. 低延迟原则：同步机制必须最小化获取和释放的延迟
2. 公平性原则：等待线程/核心应按请求顺序获得锁，避免饥饿
3. 可扩展性原则：在核心数增加时性能不显著下降
4. 节能原则：等待时减少CPU功耗，支持低功耗模式
5. 容错原则：支持锁持有者崩溃时的恢复机制

#### 10.3.2.1 总体架构

##### 自旋锁与信号量总体架构

同步机制采用层次化设计，从硬件原子指令到高级同步原语：

![图片](data:image/svg+xml,%3C%3Fxml version='1.0' encoding='UTF-8'%3F%3E%3Csvg width='1px' height='1px' viewBox='0 0 1 1' version='1.1' xmlns='http://www.w3.org/2000/svg' xmlns:xlink='http://www.w3.org/1999/xlink'%3E%3Ctitle%3E%3C/title%3E%3Cg stroke='none' stroke-width='1' fill='none' fill-rule='evenodd' fill-opacity='0'%3E%3Cg transform='translate(-249.000000, -126.000000)' fill='%23FFFFFF'%3E%3Crect x='249' y='126' width='1' height='1'%3E%3C/rect%3E%3C/g%3E%3C/g%3E%3C/svg%3E)

自旋锁与信号量总体架构解释：

1. 硬件支持层
- 必要性：需要硬件提供原子操作和内存屏障支持
	- 功能：提供原子指令、内存屏障、缓存一致性硬件支持
	- 设计目标：实现高效的原子操作和内存顺序控制
3. 原子操作层
- 必要性：需要在硬件基础上构建原子操作抽象
	- 功能：提供CAS、LL/SC、原子算术等原子操作
	- 设计目标：提供跨平台的原子操作接口
5. 同步原语层
- 必要性：需要在原子操作基础上构建同步原语
	- 功能：实现自旋锁、信号量、读写锁等同步原语
	- 设计目标：提供高效、可靠的同步机制
7. 高级接口层
- 必要性：需要为应用程序提供易用的同步接口
	- 功能：提供互斥锁、条件变量、屏障等高级接口
	- 设计目标：降低应用开发复杂度，提高开发效率

/** * 自旋锁结构定义 */typedef struct {    volatile uint32_t lock;           // 锁状态    uint32_t owner_core;              // 当前持有者核心ID    uint32_t recursion_count;         // 递归计数    spinlock_type_t type;             // 锁类型    spinlock_stats_t stats;           // 锁统计信息    uint32_t magic;                   // 魔术数字（用于验证）} spinlock_t;/** * 自旋锁统计结构 */typedef struct {    uint64_t lock_attempts;          // 锁尝试次数    uint64_t lock_success;           // 锁成功次数    uint64_t contention_count;       // 竞争次数    uint64_t max_spin_time;          // 最大自旋时间    uint64_t total_spin_time;        // 总自旋时间    uint32_t current_waiters;        // 当前等待者数量} spinlock_stats_t;/** * 初始化自旋锁 */void spinlock_init(spinlock_t *lock, spinlock_type_t type) {    if (lock == NULL) {        return;    }    lock->lock = 0;    lock->owner_core = 0xFF; // 无效核心ID    lock->recursion_count = 0;    lock->type = type;    lock->magic = SPINLOCK_MAGIC;    // 初始化统计信息    memset(&lock->stats, 0, sizeof(spinlock_stats_t));    // 内存屏障，确保初始化完成    memory_barrier();    SPINLOCK_LOG("Spinlock initialized, type=%d", type);}/** * 获取自旋锁 */void spinlock_lock(spinlock_t *lock) {    uint32_t core_id = get_cpu_id();    uint32_t expected, desired;    uint64_t start_time, spin_time;    if (lock == NULL || lock->magic != SPINLOCK_MAGIC) {        SPINLOCK_LOG("Invalid spinlock");        return;    }    // 更新统计信息    lock->stats.lock_attempts++;    // 检查递归锁（如果已经是持有者）    if (lock->owner_core == core_id) {        lock->recursion_count++;        SPINLOCK_LOG("Core %d recursive lock, count=%d",                     core_id, lock->recursion_count);        return;    }    start_time = get_system_timestamp();    // 根据锁类型选择获取策略    switch (lock->type) {        case SPINLOCK_TYPE_SIMPLE:            spin_time = acquire_simple_spinlock(lock, core_id);            break;        case SPINLOCK_TYPE_QUEUE:            spin_time = acquire_queue_spinlock(lock, core_id);            break;        case SPINLOCK_TYPE_ADAPTIVE:            spin_time = acquire_adaptive_spinlock(lock, core_id);            break;        case SPINLOCK_TYPE_READER_WRITER:            spin_time = acquire_rw_spinlock(lock, core_id, LOCK_MODE_EXCLUSIVE);            break;        default:            SPINLOCK_LOG("Unknown spinlock type: %d", lock->type);            return;    }    // 更新锁状态    lock->owner_core = core_id;    lock->recursion_count = 1;    // 更新统计信息    lock->stats.lock_success++;    lock->stats.total_spin_time += spin_time;    if (spin_time > lock->stats.max_spin_time) {        lock->stats.max_spin_time = spin_time;    }    // 内存屏障，确保锁状态对其他核心可见    memory_barrier();    SPINLOCK_LOG("Core %d acquired spinlock, spin_time=%llu ns",                 core_id, spin_time);}/** * 获取简单自旋锁 */static uint64_t acquire_simple_spinlock(spinlock_t *lock, uint32_t core_id) {    uint32_t expected, desired;    uint64_t start_time, current_time;    uint32_t spin_count = 0;    start_time = get_system_timestamp();    while (1) {        // 尝试快速获取        expected = 0;        desired = 1;        if (atomic_compare_exchange_weak(&lock->lock, &expected, desired)) {            // 获取成功            current_time = get_system_timestamp();            lock->stats.contention_count += spin_count;            return current_time - start_time;        }        // 获取失败，执行指数退避        spin_count++;        if (spin_count > SPIN_THRESHOLD) {            lock->stats.contention_count++;        }        // 指数退避等待        uint32_t delay = SPIN_BASE_DELAY * (1 << (spin_count % MAX_BACKOFF_SHIFT));        delay = MIN(delay, SPIN_MAX_DELAY);        // 等待期间检查是否值得继续自旋        current_time = get_system_timestamp();        if (current_time - start_time > SPIN_TIMEOUT_NS) {            // 自旋超时，可能切换到排队模式或休眠            SPINLOCK_LOG("Core %d spin timeout, switching strategy", core_id);            if (lock->type == SPINLOCK_TYPE_ADAPTIVE) {                return acquire_queue_spinlock(lock, core_id);            }        }        // 执行退避延迟        for (uint32_t i = 0; i < delay; i++) {            cpu_relax(); // 架构相关的放松指令        }    }}/** * 获取排队自旋锁 */static uint64_t acquire_queue_spinlock(spinlock_t *lock, uint32_t core_id) {    queue_node_t *node = allocate_queue_node();    queue_node_t *prev;    uint64_t start_time, current_time;    if (node == NULL) {        SPINLOCK_LOG("Failed to allocate queue node");        return acquire_simple_spinlock(lock, core_id); // 回退到简单自旋锁    }    // 初始化节点    node->core_id = core_id;    node->next = NULL;// 初始时锁定自己的节点
    start_time = get_system_timestamp();    // 将节点加入队列尾部    prev = atomic_exchange(&lock->tail, node);    if (prev != NULL) {        // 队列不为空，链接到前一个节点        prev->next = node;        // 等待前一个节点释放锁        while (node->locked) {            cpu_relax();        }    } else {        // 队列为空，直接获取锁        lock->lock = 1;        node->locked = 0;    }    // 保存当前节点到线程本地存储，用于释放时找到自己的节点    set_thread_local_node(node);    current_time = get_system_timestamp();    return current_time - start_time;}/** * 释放自旋锁 */void spinlock_unlock(spinlock_t *lock) {    uint32_t core_id = get_cpu_id();    queue_node_t *node;    if (lock == NULL || lock->magic != SPINLOCK_MAGIC) {        SPINLOCK_LOG("Invalid spinlock");        return;    }    // 检查是否是锁的持有者    if (lock->owner_core != core_id) {        SPINLOCK_LOG("Core %d attempted to unlock spinlock owned by core %d",                    core_id, lock->owner_core);        return;    }    // 减少递归计数    lock->recursion_count--;    // 如果还有递归计数，不释放锁    if (lock->recursion_count > 0) {        SPINLOCK_LOG("Core %d recursive unlock, remaining count=%d",                    core_id, lock->recursion_count);        return;    }    // 根据锁类型选择释放策略    switch (lock->type) {        case SPINLOCK_TYPE_SIMPLE:            release_simple_spinlock(lock);            break;        case SPINLOCK_TYPE_QUEUE:            node = get_thread_local_node();            release_queue_spinlock(lock, node);            break;        case SPINLOCK_TYPE_ADAPTIVE:            // 自适应锁根据当前状态决定释放策略            if (lock->stats.current_waiters > ADAPTIVE_QUEUE_THRESHOLD) {                node = get_thread_local_node();                if (node != NULL) {                    release_queue_spinlock(lock, node);                } else {                    release_simple_spinlock(lock);                }            } else {                release_simple_spinlock(lock);            }            break;        case SPINLOCK_TYPE_READER_WRITER:            release_rw_spinlock(lock, core_id);            break;        default:            SPINLOCK_LOG("Unknown spinlock type: %d", lock->type);            return;    }    // 清除持有者信息    lock->owner_core = 0xFF;    // 内存屏障，确保锁释放对其他核心可见    memory_barrier();    SPINLOCK_LOG("Core %d released spinlock", core_id);}/** * 释放简单自旋锁 */static void release_simple_spinlock(spinlock_t *lock) {    // 内存屏障确保临界区操作在锁释放前完成    memory_barrier();    // 清除锁标志    atomic_store(&lock->lock, 0);}/** * 释放排队自旋锁 */static void release_queue_spinlock(spinlock_t *lock, queue_node_t *node) {    queue_node_t *next;    // 检查是否有下一个等待者    next = node->next;    if (next == NULL) {        // 没有等待者，尝试将尾指针设置为NULL        queue_node_t *tail = node;        if (atomic_compare_exchange_strong(&lock->tail, &tail, NULL)) {            // 成功设置尾指针为NULL，直接清除锁            memory_barrier();            atomic_store(&lock->lock, 0);        } else {            // 有新的节点加入，等待它链接            while (node->next == NULL) {                cpu_relax();            }            next = node->next;            // 传递锁给下一个节点            memory_barrier();            next->locked = 0;        }    } else {        // 直接传递锁给下一个节点        memory_barrier();        next->locked = 0;    }    // 释放节点内存（可选，可以缓存重用）    free_queue_node(node);}

/** * 信号量结构定义 */typedef struct {    volatile int32_t count;           // 信号量计数器    uint32_t max_count;               // 最大计数值    spinlock_t lock;                  // 内部保护锁    wait_queue_t wait_queue;          // 等待队列    semaphore_type_t type;            // 信号量类型    semaphore_stats_t stats;          // 统计信息    uint32_t magic;                   // 魔术数字} semaphore_t;/** * 等待队列结构 */typedef struct {    wait_node_t *head;                // 队列头    wait_node_t *tail;                // 队列尾    uint32_t size;                    // 队列大小} wait_queue_t;/** * 等待节点结构 */typedef struct wait_node {    uint32_t core_id;                 // 等待核心ID    uint32_t priority;                // 等待优先级    volatile uint32_t signaled;       // 信号标志    struct wait_node *next;           // 下一个节点} wait_node_t;/** * 初始化信号量 */void semaphore_init(semaphore_t *sem, int32_t initial_count,                    uint32_t max_count, semaphore_type_t type) {    if (sem == NULL) {        return;    }    // 参数检查    if (initial_count < 0 || initial_count > max_count) {        SEMAPHORE_LOG("Invalid initial count: %d, max: %d",                      initial_count, max_count);        return;    }    sem->count = initial_count;    sem->max_count = max_count;    sem->type = type;    sem->magic = SEMAPHORE_MAGIC;    // 初始化内部锁    spinlock_init(&sem->lock, SPINLOCK_TYPE_SIMPLE);    // 初始化等待队列    sem->wait_queue.head = NULL;    sem->wait_queue.tail = NULL;    sem->wait_queue.size = 0;    // 初始化统计信息    memset(&sem->stats, 0, sizeof(semaphore_stats_t));    // 内存屏障，确保初始化完成    memory_barrier();    SEMAPHORE_LOG("Semaphore initialized: count=%d, max=%d, type=%d",                 initial_count, max_count, type);}/** * 等待信号量 */semaphore_result_t semaphore_wait(semaphore_t *sem, uint32_t timeout_ms) {    wait_node_t node;    uint64_t start_time, current_time;    semaphore_result_t result = SEMAPHORE_OK;    if (sem == NULL || sem->magic != SEMAPHORE_MAGIC) {        SEMAPHORE_LOG("Invalid semaphore");        return SEMAPHORE_INVALID;    }    // 更新统计信息    sem->stats.wait_attempts++;    start_time = get_system_timestamp();    // 快速路径：如果计数器大于0，直接减少    if (atomic_load(&sem->count) > 0) {        int32_t old_count = atomic_fetch_sub(&sem->count, 1);        if (old_count > 0) {            // 成功获取信号量            sem->stats.wait_success++;            return SEMAPHORE_OK;        } else {            // 竞争条件：计数器实际上为0，恢复计数器            atomic_fetch_add(&sem->count, 1);        }    }    // 慢速路径：需要等待    // 初始化等待节点    node.core_id = get_cpu_id();    node.priority = get_current_priority();    node.signaled = 0;    node.next = NULL;    // 获取内部锁    spinlock_lock(&sem->lock);    // 再次检查计数器（避免竞争）    if (sem->count > 0) {        sem->count--;        spinlock_unlock(&sem->lock);        sem->stats.wait_success++;        return SEMAPHORE_OK;    }    // 将节点加入等待队列    if (sem->wait_queue.tail == NULL) {        // 队列为空        sem->wait_queue.head = &node;        sem->wait_queue.tail = &node;    } else {        // 插入队列尾部        sem->wait_queue.tail->next = &node;        sem->wait_queue.tail = &node;    }    sem->wait_queue.size++;    // 更新统计信息    sem->stats.current_waiters = sem->wait_queue.size;    spinlock_unlock(&sem->lock);    // 等待信号    while (node.signaled == 0) {        // 检查超时        if (timeout_ms != SEMAPHORE_WAIT_INFINITE) {            current_time = get_system_timestamp();            if (current_time - start_time > (timeout_ms * 1000)) {                // 超时，从队列中移除节点                spinlock_lock(&sem->lock);                // 检查节点是否还在队列中（可能在此期间被唤醒）                if (node.signaled == 0) {                    // 从队列中移除节点                    remove_node_from_queue(sem, &node);                    result = SEMAPHORE_TIMEOUT;                } else {                    result = SEMAPHORE_OK;                }                spinlock_unlock(&sem->lock);                break;            }        }        // 低功耗等待        cpu_relax();        // 检查信号标志（使用内存屏障确保可见性）        memory_barrier();    }    if (result == SEMAPHORE_OK) {        sem->stats.wait_success++;    } else {        sem->stats.wait_timeouts++;    }    // 计算等待时间    current_time = get_system_timestamp();    uint64_t wait_time = current_time - start_time;    sem->stats.total_wait_time += wait_time;    if (wait_time > sem->stats.max_wait_time) {        sem->stats.max_wait_time = wait_time;    }    SEMAPHORE_LOG("Semaphore wait %s, wait_time=%llu ns",                  result == SEMAPHORE_OK ? "success" : "timeout", wait_time);    return result;}/** * 发布信号量 */semaphore_result_t semaphore_post(semaphore_t *sem) {    wait_node_t *node_to_wake = NULL;    if (sem == NULL || sem->magic != SEMAPHORE_MAGIC) {        SEMAPHORE_LOG("Invalid semaphore");        return SEMAPHORE_INVALID;    }    // 快速路径：如果没有等待者，直接增加计数器    if (sem->wait_queue.size == 0) {        int32_t old_count = atomic_load(&sem->count);        if (old_count < sem->max_count) {            atomic_fetch_add(&sem->count, 1);            sem->stats.post_count++;            return SEMAPHORE_OK;        } else {            SEMAPHORE_LOG("Semaphore count would exceed max");            return SEMAPHORE_OVERFLOW;        }    }    // 慢速路径：有等待者，唤醒一个    spinlock_lock(&sem->lock);    // 再次检查（避免竞争）    if (sem->wait_queue.size == 0) {        // 等待者在检查后消失了，直接增加计数器        if (sem->count < sem->max_count) {            sem->count++;            spinlock_unlock(&sem->lock);            sem->stats.post_count++;            return SEMAPHORE_OK;        } else {            spinlock_unlock(&sem->lock);            return SEMAPHORE_OVERFLOW;        }    }    // 根据信号量类型选择唤醒策略    switch (sem->type) {        case SEMAPHORE_TYPE_FIFO:            // FIFO：唤醒队列头部的等待者            node_to_wake = sem->wait_queue.head;            break;        case SEMAPHORE_TYPE_PRIORITY:            // 优先级：唤醒优先级最高的等待者            node_to_wake = find_highest_priority_node(sem);            break;        case SEMAPHORE_TYPE_BINARY:            // 二进制信号量：只唤醒一个等待者            node_to_wake = sem->wait_queue.head;            break;        default:            SEMAPHORE_LOG("Unknown semaphore type: %d", sem->type);            spinlock_unlock(&sem->lock);            return SEMAPHORE_ERROR;    }    if (node_to_wake != NULL) {        // 从队列中移除节点        remove_node_from_queue(sem, node_to_wake);        // 设置信号标志        memory_barrier();        node_to_wake->signaled = 1;        // 内存屏障确保信号标志对其他核心可见        memory_barrier();        sem->stats.wakeup_count++;        sem->stats.current_waiters = sem->wait_queue.size;    }    spinlock_unlock(&sem->lock);    SEMAPHORE_LOG("Semaphore posted, woke up core %d",                  node_to_wake ? node_to_wake->core_id : -1);    return SEMAPHORE_OK;}/** * 从队列中移除节点 */static void remove_node_from_queue(semaphore_t *sem, wait_node_t *node) {    wait_node_t *prev = NULL;    wait_node_t *current = sem->wait_queue.head;    // 查找节点    while (current != NULL && current != node) {        prev = current;        current = current->next;    }    if (current == NULL) {        // 节点不在队列中（可能已经被移除）        return;    }    // 从队列中移除节点    if (prev == NULL) {        // 节点是队列头部        sem->wait_queue.head = node->next;    } else {        prev->next = node->next;    }    // 如果节点是队列尾部，更新尾指针    if (node == sem->wait_queue.tail) {        sem->wait_queue.tail = prev;    }    sem->wait_queue.size--;}/** * 查找优先级最高的节点 */static wait_node_t *find_highest_priority_node(semaphore_t *sem) {    wait_node_t *current = sem->wait_queue.head;    wait_node_t *highest = NULL;    uint32_t highest_priority = 0xFFFFFFFF; // 最低优先级    while (current != NULL) {        if (current->priority < highest_priority) {            highest_priority = current->priority;            highest = current;        }        current = current->next;    }    return highest;}
